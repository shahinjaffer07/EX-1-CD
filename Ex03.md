## PROGRAM
arth.l
```
%{
#include "y.tab.h"
#include <stdio.h>
%}

%%

[ \t]+      ;

"="          { printf("\nOperator is EQUAL"); return '='; }
"+"          { printf("\nOperator is PLUS"); return PLUS; }
"-"          { printf("\nOperator is MINUS"); return MINUS; }
"*"          { printf("\nOperator is MULTIPLICATION"); return MULTIPLICATION; }
"/"          { printf("\nOperator is DIVISION"); return DIVISION; }

[A-Za-z][A-Za-z0-9]*   { printf("\nIdentifier is %s", yytext); return ID; }

\n          { }
.           { return yytext[0]; }

%%

int yywrap(void) { return 1; }
```
arth.y
```
%{
#include <stdio.h>
void yyerror(char *s);
%}

%token ID PLUS MINUS MULTIPLICATION DIVISION

%%

statement:
      ID '=' E { printf("\nValid arithmetic expression\n"); }
;

E:
      E PLUS ID
    | E MINUS ID
    | E MULTIPLICATION ID
    | E DIVISION ID
    | ID
;

%%

int main(void) {
    yyparse();
    return 0;
}

void yyerror(char *s) {
    printf("Error: %s\n", s);
}
```
