```
[aA]    { return A; }
[bB]    { return B; }

[ \t]+  { /* skip spaces/tabs */ }
.       { /* ignore other characters */ }
```

anb.y
```
%{
#include <stdio.h>
#include <stdlib.h>

/* prototype for lexer */
int yylex(void);
void yyerror(const char *msg);
%}

/* return integer counts from nonterminals */
%union {
    int num;
}

%token A B NL
%type <num> As

%%

input:
      stmt NL   { /* accepted inside stmt */ }
    | stmt      { /* allow no final newline too */ }
    ;

stmt:
      As B      {
                   if ($1 >= 10) {
                       printf("valid string\n");
                       exit(0);
                   } else {
                       printf("invalid string\n");
                       exit(0);
                   }
                 }
    ;

/* As returns the number of A tokens */
As:
      As A      { $$ = $1 + 1; }
    | A         { $$ = 1; }
    ;

%%

void yyerror(const char *msg) {
    /* if parser calls error, we treat as invalid */
    (void)msg;
    printf("invalid string\n");
    exit(0);
}

int main(void) {
    /* prompt optional */
    printf("enter the string\n");
    yyparse();
    return 0;
}
```
