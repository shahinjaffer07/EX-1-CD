## AIM   
To write a C program to implement a symbol table.

## ALGORITHM
1.	Start the program.
2.	Get the input from the user with the terminating symbol ‘$’.
3.	Allocate memory for the variable by dynamic memory allocation function.
4.	If the next character of the symbol is an operator then only the memory is allocated.
5.	While reading, the input symbol is inserted into symbol table along with its memory address.
6.	The steps are repeated till ‘$’ is reached.
7.	To reach a variable, enter the variable to be searched and symbol table has been checked for corresponding variable, the variable along with its address is displayed as result.
8.	Stop the program. 

## PROGRAM
```
#include <stdio.h>
#include <ctype.h>
#include <string.h>
#include <stdlib.h>

#define MAX 100

int main() {
    char expr[MAX], c, search;
    char symbols[50];     // store symbols
    void *addr[50];       // store addresses
    int i = 0, k = 0, found = 0;

    printf("Enter the expression terminated by $: ");

    // Read characters until '$'
    while ((c = getchar()) != '$' && i < MAX - 1) {
        expr[i++] = c;
    }
    expr[i] = '\0';

    printf("\nGiven Expression: %s\n", expr);

    printf("\nSymbol Table\n");
    printf("Symbol\tAddress\t\tType\n");

    for (int j = 0; j < i; j++) {
        c = expr[j];

        // If alphabet → possible identifier
        if (isalpha((unsigned char)c)) {

            // Check if next char is operator or end
            if (j == i - 1 || expr[j+1] == '+' || expr[j+1] == '-' || 
                expr[j+1] == '*' || expr[j+1] == '/' || expr[j+1] == '=') {

                void *p = malloc(sizeof(char)); // allocate memory
                if (!p) {
                    printf("Memory allocation failed!\n");
                    return 1;
                }

                symbols[k] = c;
                addr[k] = p;

                printf("%c\t%p\tidentifier\n", c, p);
                k++;
            }
        }
    }

    // Search part
    printf("\nEnter symbol to search: ");
    getchar(); // consume newline
    search = getchar();

    for (int j = 0; j < k; j++) {
        if (symbols[j] == search) {
            printf("\nSymbol Found\n");
            printf("%c @ address %p\n", search, addr[j]);
            found = 1;
            break;
        }
    }

    if (!found)
        printf("\nSymbol Not Found\n");

    // Free memory
    for (int j = 0; j < k; j++) {
        free(addr[j]);
    }

    return 0;
}

```
